## Educación


```{.r .cell-code}
library(typstcv)
library(tidyverse)

df_cv <- readxl::read_excel('harry_potter_cv_es.xlsx') |> 
  mutate(start = as_date(start),
end = as_date(end)) |> 
  drop_na(start)

df_cv |> 
  filter(section == 'educacion') |>
  format_date(end = "end", sort_by = "start") |>
  resume_entry(title = "role",
               description = "institution",
               details = "descripcion")
```
